 * NOTESCSBS | Institutional Welcome Broadcast

 * This script automates the delivery of a high-fidelity Welcome Brief 
 * when a student enrolls in the Academic Shift Registry.

 * Trigger: On Form Submit
 * Setup: Deploy as a trigger in Google Apps Script associated with your response sheet.

function onFormSubmit(e) {
  // Capture captured email (adjust index if email is not in the first column after timestamp)
  const email = e.values[1]; 
  
  // Check for duplicates in the spreadsheet
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const data = sheet.getDataRange().getValues();
  
  let count = 0;
  // Assuming email is in the second column (index 1)
  for (let i = 0; i < data.length; i++) {
    if (data[i][1] === email) {
      count++;
    }
  }
  
  // If count is > 1, it's a duplicate submission (the current one makes it at least 1)
  if (count > 1) {
    // Delete the duplicate row to keep the spreadsheet clean
    sheet.deleteRow(e.range.getRow());
    // Stop execution so we don't send the email again
    return;
